// watchPosition मुळे तुझे लोकेशन हलले की तो लगेच पकडतो navigator.geolocation.watchPosition((position) => { const lat = position.coords.latitude; const lon = position.coords.longitude; // डेटाबेसमध्ये सतत अपडेट करणे database.ref('mylocation/').set({ lat: lat, lon: lon, timestamp: Date.now() }); document.getElementById('status').innerText = "Tracking: " + lat + ", " + lon; }, (err) => { console.error(err); }, { enableHighAccuracy: true, // एकदम एक्झॅक्ट लोकेशनसाठी maximumAge: 0, timeout: 5000 });
